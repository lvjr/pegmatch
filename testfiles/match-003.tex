\documentclass{article}
\input{regression-test}
\SetSpegTracing{+match}
\begin{document}

\START

\BEGINTEST{multiple operators}
\SetSpeg\lTmpaSpeg{ !\SpegP{cd} * \SpegQ{1} }
\MatchSpeg\lTmpaSpeg{}
\MatchSpeg\lTmpaSpeg{c}
\MatchSpeg\lTmpaSpeg{cde}
\MatchSpeg\lTmpaSpeg{abc}
\ENDTEST

\BEGINTEST{multiple operators}
\SetSpeg\lTmpaSpeg{ ! \SpegR{an} * \SpegQ{1} }
\MatchSpeg\lTmpaSpeg{}
\MatchSpeg\lTmpaSpeg{cc}
\MatchSpeg\lTmpaSpeg{nn}
\MatchSpeg\lTmpaSpeg{zz}
\ENDTEST

\BEGINTEST{multiple operators}
\SetSpeg\lTmpaSpeg{ ! \SpegR{AZaz} * \SpegQ{1} }
\MatchSpeg\lTmpaSpeg{}
\MatchSpeg\lTmpaSpeg{[]}
\MatchSpeg\lTmpaSpeg{X2}
\MatchSpeg\lTmpaSpeg{12}
\MatchSpeg\lTmpaSpeg{x2}
\MatchSpeg\lTmpaSpeg{12}
\ENDTEST

\BEGINTEST{multiple operators}
\SetSpeg\lTmpaSpeg{ ! \SpegS{asc} * \SpegQ{1} }
\MatchSpeg\lTmpaSpeg{}
\MatchSpeg\lTmpaSpeg{aa}
\MatchSpeg\lTmpaSpeg{bb}
\MatchSpeg\lTmpaSpeg{cc}
\ENDTEST

\BEGINTEST{multiple operators with errors}
\SetSpeg\lTmpaSpeg{ \SpegP{ab} * \SpegQ{1} * \SpegP{de} / \SpegP{12} * \SpegQ{2} * \SpegP{56} }
\MatchSpeg\lTmpaSpeg{}
\MatchSpeg\lTmpaSpeg{*****}
\MatchSpeg\lTmpaSpeg{abxyz}
\MatchSpeg\lTmpaSpeg{ab0de}
\MatchSpeg\lTmpaSpeg{12xyz}
\MatchSpeg\lTmpaSpeg{12xy56}
\ENDTEST

\BEGINTEST{multiple operators with errors}
\SetSpeg\lTmpaSpeg{ \SpegR{AZ} ^ {4} / \SpegR{09} ^ 3 * \SpegS{abcd} }
\MatchSpeg\lTmpaSpeg{}
\MatchSpeg\lTmpaSpeg{ABC}
\MatchSpeg\lTmpaSpeg{ABCd}
\MatchSpeg\lTmpaSpeg{ABCD}
\MatchSpeg\lTmpaSpeg{12x}
\MatchSpeg\lTmpaSpeg{159x}
\MatchSpeg\lTmpaSpeg{159c}
\ENDTEST

\end{document}
