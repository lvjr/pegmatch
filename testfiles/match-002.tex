\START

\BEGINTEST{choice operator}
\SetTpeg\lTmpaTpeg{\TpegP{ab}/\TpegP{mn}/\TpegP{uv}}
\MatchTpeg\lTmpaTpeg{cdef}
\MatchTpeg\lTmpaTpeg{abcd}
\MatchTpeg\lTmpaTpeg{uvwx}
\ENDTEST

\BEGINTEST{concat operator}
\SetTpeg\lTmpaTpeg{\TpegP{ab}*\TpegP{mn}*\TpegP{uv}}
\MatchTpeg\lTmpaTpeg{cdef}
\MatchTpeg\lTmpaTpeg{abcd}
\MatchTpeg\lTmpaTpeg{abmnxy}
\MatchTpeg\lTmpaTpeg{abmnuv}
\MatchTpeg\lTmpaTpeg{abmnuvxy}
\ENDTEST

\BEGINTEST{not predicate operator}
\SetTpeg\lTmpaTpeg{ ! \TpegP{ddd} }
\MatchTpeg\lTmpaTpeg{dddd}
\MatchTpeg\lTmpaTpeg{eeee}
\ENDTEST

\BEGINTEST{and predicate operator}
\SetTpeg\lTmpaTpeg{ & \TpegP{eee} }
\MatchTpeg\lTmpaTpeg{dddd}
\MatchTpeg\lTmpaTpeg{eeee}
\ENDTEST

\BEGINTEST{repeat operator}
\SetTpeg\lTmpaTpeg{ \TpegP{test} ^ 2 }
\MatchTpeg\lTmpaTpeg{temp}
\MatchTpeg\lTmpaTpeg{test}
\MatchTpeg\lTmpaTpeg{testtemp}
\MatchTpeg\lTmpaTpeg{testtest}
\MatchTpeg\lTmpaTpeg{testtesttemp}
\MatchTpeg\lTmpaTpeg{testtesttest}
\MatchTpeg\lTmpaTpeg{testtesttesttemp}
\ENDTEST

\BEGINTEST{repeat operator}
\SetTpeg\lTmpaTpeg{ \TpegP{test} ^ 0 }
\MatchTpeg\lTmpaTpeg{temp}
\MatchTpeg\lTmpaTpeg{test}
\MatchTpeg\lTmpaTpeg{testtemp}
\MatchTpeg\lTmpaTpeg{testtest}
\MatchTpeg\lTmpaTpeg{testtesttemp}
\MatchTpeg\lTmpaTpeg{testtesttest}
\MatchTpeg\lTmpaTpeg{testtesttesttemp}
\ENDTEST

\BEGINTEST{repeat operator}
\SetTpeg\lTmpaTpeg{ \TpegP{test} ^ {-1} }
\MatchTpeg\lTmpaTpeg{temp}
\MatchTpeg\lTmpaTpeg{test}
\MatchTpeg\lTmpaTpeg{testtemp}
\MatchTpeg\lTmpaTpeg{testtest}
\ENDTEST

\BEGINTEST{repeat operator}
\SetTpeg\lTmpaTpeg{ \TpegP{test} ^ {-2} }

\MatchTpeg\lTmpaTpeg{test}
\MatchTpeg\lTmpaTpeg{testtemp}
\MatchTpeg\lTmpaTpeg{testtest}
\MatchTpeg\lTmpaTpeg{testtesttemp}
\MatchTpeg\lTmpaTpeg{testtesttest}
\ENDTEST
