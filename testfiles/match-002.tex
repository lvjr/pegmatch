\documentclass{article}
\input{regression-test}
\SetSpegTracing{+match}
\begin{document}

\START

\BEGINTEST{choice operator}
\SetSpeg\lTmpaSpeg{\SpegP{ab}/\SpegP{mn}/\SpegP{uv}}
\MatchSpeg\lTmpaSpeg{cdef}
\MatchSpeg\lTmpaSpeg{abcd}
\MatchSpeg\lTmpaSpeg{uvwx}
\ENDTEST

\BEGINTEST{concat operator}
\SetSpeg\lTmpaSpeg{\SpegP{ab}*\SpegP{mn}*\SpegP{uv}}
\MatchSpeg\lTmpaSpeg{cdef}
\MatchSpeg\lTmpaSpeg{abcd}
\MatchSpeg\lTmpaSpeg{abmnxy}
\MatchSpeg\lTmpaSpeg{abmnuv}
\MatchSpeg\lTmpaSpeg{abmnuvxy}
\ENDTEST

\BEGINTEST{not predicate operator}
\SetSpeg\lTmpaSpeg{ ! \SpegP{ddd} }
\MatchSpeg\lTmpaSpeg{dddd}
\MatchSpeg\lTmpaSpeg{eeee}
\ENDTEST

\BEGINTEST{and predicate operator}
\SetSpeg\lTmpaSpeg{ & \SpegP{eee} }
\MatchSpeg\lTmpaSpeg{dddd}
\MatchSpeg\lTmpaSpeg{eeee}
\ENDTEST

\BEGINTEST{repeat operator}
\SetSpeg\lTmpaSpeg{ \SpegP{test} ^ 2 }
\MatchSpeg\lTmpaSpeg{temp}
\MatchSpeg\lTmpaSpeg{test}
\MatchSpeg\lTmpaSpeg{testtemp}
\MatchSpeg\lTmpaSpeg{testtest}
\MatchSpeg\lTmpaSpeg{testtesttemp}
\MatchSpeg\lTmpaSpeg{testtesttest}
\MatchSpeg\lTmpaSpeg{testtesttesttemp}
\ENDTEST

\BEGINTEST{repeat operator}
\SetSpeg\lTmpaSpeg{ \SpegP{test} ^ 0 }
\MatchSpeg\lTmpaSpeg{temp}
\MatchSpeg\lTmpaSpeg{test}
\MatchSpeg\lTmpaSpeg{testtemp}
\MatchSpeg\lTmpaSpeg{testtest}
\MatchSpeg\lTmpaSpeg{testtesttemp}
\MatchSpeg\lTmpaSpeg{testtesttest}
\MatchSpeg\lTmpaSpeg{testtesttesttemp}
\ENDTEST

\BEGINTEST{repeat operator}
\SetSpeg\lTmpaSpeg{ \SpegP{test} ^ {-1} }
\MatchSpeg\lTmpaSpeg{temp}
\MatchSpeg\lTmpaSpeg{test}
\MatchSpeg\lTmpaSpeg{testtemp}
\MatchSpeg\lTmpaSpeg{testtest}
\ENDTEST

\BEGINTEST{repeat operator}
\SetSpeg\lTmpaSpeg{ \SpegP{test} ^ {-2} }
\MatchSpeg\lTmpaSpeg{temp}
\MatchSpeg\lTmpaSpeg{test}
\MatchSpeg\lTmpaSpeg{testtemp}
\MatchSpeg\lTmpaSpeg{testtest}
\MatchSpeg\lTmpaSpeg{testtesttemp}
\MatchSpeg\lTmpaSpeg{testtesttest}
\ENDTEST

\end{document}
